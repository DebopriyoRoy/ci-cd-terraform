pipeline{
    agent any
    options {
        buildDiscarder logRotator(artifactDaysToKeepStr: '', artifactNumToKeepStr: '', daysToKeepStr: '30', numToKeepStr: '2')
    }
    tools{
        maven 'Maven'
    }
    environment{
        cred = credentials('aws-key')
    }
    stages{
        stage('Checkout stage'){
            steps{
                checkout scmGit(branches: [[name: '*/main']], extensions: [], userRemoteConfigs: [[url: 'https://github.com/DebopriyoRoy/ci-cd-terraform.git']])
            }
        }
        stage('sonar test') {
            steps{
                script{
                    def mvn = tool 'Maven'
                    withSonarQubeEnv(installationName: 'sonarqube-server') {
                        //sh "${mvn}/bin/mvn clean verify sonar:sonar -Dsonar.projectKey=finalproject -Dsonar.projectName='finalproject'"
                          sh "${mvn}/bin/mvn clean verify sonar:sonar -Dsonar.projectKey=trialproject -Dsonar.projectName='trialproject'"

                    }
                }
            }
        }
        stage('maven build'){
            steps{
                sh 'mvn package'
            }
        }
        stage('Nexus Push'){
            steps{
                nexusArtifactUploader(
                    nexusVersion: 'nexus3',
                    protocol: 'http',
                    nexusUrl: '10.0.1.53:8081',
                    groupId: 'addressbook',
                    version: '2.0-SNAPSHOT',
                    repository: 'maven-snapshots',
                    credentialsId: 'nexus',
                    artifacts: [
                        [artifactId: 'trialproject',
                        classifier: '',
                        file: 'target/addressbook-2.0.war',
                        type: 'war']
                     ]
                )
            }
        }
    stage('Docker build'){
            steps{
                sh 'docker build -t cicd-jenkins .'
            }
        }
    stage('DockerHub push'){
            steps{
                sh 'aws ecr get-login-password --region us-east-1 | docker login --username AWS --password-stdin 557270420072.dkr.ecr.us-east-1.amazonaws.com'
                //sh "docker tag cicd-jenkins:latest 557270420072.dkr.ecr.us-east-1.amazonaws.com/cicd-jenkins:latest"
                //sh "docker push 557270420072.dkr.ecr.us-east-1.amazonaws.com/cicd-jenkins:latest"
                sh "docker tag cicd-jenkins:latest 557270420072.dkr.ecr.us-east-1.amazonaws.com/cicd-jenkins:${BUILD_NUMBER}"
                sh "docker push 557270420072.dkr.ecr.us-east-1.amazonaws.com/cicd-jenkins:${BUILD_NUMBER}"
            }
        }
    stage('Kubernetes deploy'){
            steps{
                sh 'aws eks update-kubeconfig --region us-east-1 --name charming-rock-ant' // (kubernetes cluster name)
                sh 'kubectl apply -f Application.yaml'
            }
        }
    }
    post {
        always {
            echo "Job is completed"
        }
        success {
            echo "success"
        }
        failure {
            echo "failed"
        }
    }
}
